# i18n: Korean & English Multilingual Support — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Korean/English multilingual support using next-intl with locale-prefixed URLs (`/en/*`, `/ko/*`) and automatic browser language detection.

**Architecture:** next-intl wraps the Next.js config and middleware to handle locale detection/routing. All app pages move under `src/app/[locale]/`. Server components use `getTranslations()`, client components use `useTranslations()`. `generateStaticParams` + `setRequestLocale` in the locale layout ensure static rendering works for all locales.

**Tech Stack:** `next-intl` (new), Next.js 16 App Router, TypeScript, Vitest, Playwright

```ts
import { getRequestConfig } from 'next-intl/server';
import { notFound } from 'next/navigation';
import { routing } from './routing';

export default getRequestConfig(async ({ requestLocale }) => {
  const locale = await requestLocale;

  if (!locale || !routing.locales.includes(locale as 'en' | 'ko')) {
    notFound();
  }

  return {
    locale,
    messages: (await import(`../../messages/${locale}.json`)).default,
  };
});
```

### Task 3: Replace middleware with next-intl routing

**Files:**

- Modify: `src/middleware.ts`
- Modify: `src/middleware.test.ts`

**Intentional behavior change:** The old middleware validated blog slugs and app paths, redirecting any unrecognized path (e.g., `/blog/nonexistent-post`, `/invalid-path`) back to `/`. The new middleware does **not** validate paths — it only detects locale and redirects to the locale-prefixed equivalent. An unknown path like `/blog/nonexistent-post` now lands at `/en/blog/nonexistent-post`, where `notFound()` in `blog/[slug]/page.tsx` produces a 404. This is intentional: path validation belongs in page components, not the routing layer.

`src/middleware.test.ts` tests the old redirect-to-root behavior and must be completely rewritten for the new locale-detection behavior.

- [ ] **Step 1: Replace `src/middleware.ts`**

```ts
import createMiddleware from 'next-intl/middleware';
import { routing } from './i18n/routing';

/**
 * Locale-aware middleware: detects browser language from accept-language header
 * and redirects non-locale paths to the appropriate locale prefix.
 * - / → /en or /ko
 * - /blog → /en/blog or /ko/blog
 * - /blog/some-post → /en/blog/some-post or /ko/blog/some-post
 * - /en/*, /ko/* → pass through
 * - /api/*, /_next/*, static files → skipped automatically
 *
 * TODO: Rename to proxy.ts when @opennextjs/cloudflare supports Next.js 16 proxy convention
 * @see https://github.com/opennextjs/opennextjs-cloudflare/issues/962
 */
export default createMiddleware(routing);

export const config = {
  matcher: ['/((?!api|_next/static|_next/image|favicon.ico|.*\\..*).*)'],
};
```

- [ ] **Step 2: Rewrite `src/middleware.test.ts`**

```ts
import { NextRequest } from 'next/server';
import middleware from './middleware';

const createRequest = (
  pathname: string,
  acceptLanguage = 'en-US',
): NextRequest =>
  new NextRequest(new URL(pathname, 'http://localhost:3000'), {
    headers: { 'accept-language': acceptLanguage },
  });

describe('middleware', () => {
  describe('[locale detection] should redirect to locale-prefixed URL', () => {
    it('should redirect / to /en when accept-language is en-US', () => {
      // Arrange
      const request = createRequest('/', 'en-US');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(307);
      expect(response.headers.get('location')).toContain('/en');
    });

    it('should redirect / to /ko when accept-language is ko-KR', () => {
      // Arrange
      const request = createRequest('/', 'ko-KR');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(307);
      expect(response.headers.get('location')).toContain('/ko');
    });

    it('should redirect / to /en for unsupported locale fr-FR', () => {
      // Arrange
      const request = createRequest('/', 'fr-FR');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(307);
      expect(response.headers.get('location')).toContain('/en');
    });

    it('should redirect /blog to /ko/blog when accept-language is ko', () => {
      // Arrange
      const request = createRequest('/blog', 'ko');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(307);
      expect(response.headers.get('location')).toContain('/ko/blog');
    });

    it('should redirect /blog/some-post to /ko/blog/some-post when accept-language is ko', () => {
      // Arrange
      const request = createRequest('/blog/some-post', 'ko');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(307);
      expect(response.headers.get('location')).toContain('/ko/blog/some-post');
    });
  });

  describe('[pass-through] should not redirect locale-prefixed paths', () => {
    it('should pass through /en/blog without redirect', () => {
      // Arrange
      const request = createRequest('/en/blog', 'en-US');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(200);
    });

    it('should pass through /ko/blog without redirect', () => {
      // Arrange
      const request = createRequest('/ko/blog', 'ko');

      // Act
      const response = middleware(request);

      // Assert
      expect(response.status).toBe(200);
    });
  });
});
```

`usePathname` and `useRouter` from `@/i18n/navigation` are locale-aware: `usePathname()` returns locale-stripped paths (e.g., `/blog` not `/en/blog`), and `router.push('/blog')` automatically navigates to `/en/blog` or `/ko/blog` based on current locale.

**Why RetroOS and PostList must also be updated:**

- `RetroOS` does `const activeAppId = pathname.split('/')[1] || ''` to determine which window to render `children` into. With raw `next/navigation` pathname, `/en/blog` would yield `activeAppId = 'en'`, not `'blog'`, breaking window rendering.

```ts
'use client';

import { useEffect, useRef, useState, useCallback } from 'react';
import { usePathname, useRouter } from '@/i18n/navigation';

const BLOG_PATH_PREFIX = '/blog';

export interface BlogNavigation {
  canGoBack: boolean;
  canGoForward: boolean;
  goBack: () => void;
  goForward: () => void;
}

export const useBlogNavigation = (): BlogNavigation => {
  const pathname = usePathname();
  const router = useRouter();

  const historyRef = useRef<string[]>([]);
  const indexRef = useRef(-1);
  const isInternalNavRef = useRef(false);

  const [canGoBack, setCanGoBack] = useState(false);
  const [canGoForward, setCanGoForward] = useState(false);

  useEffect(() => {
    if (!pathname.startsWith(BLOG_PATH_PREFIX)) {
      return;
    }

    if (isInternalNavRef.current) {
      isInternalNavRef.current = false;
      return;
    }

    const newIndex = indexRef.current + 1;
    historyRef.current = [...historyRef.current.slice(0, newIndex), pathname];
    indexRef.current = newIndex;

    setCanGoBack(newIndex > 0);
    setCanGoForward(false);
  }, [pathname]);

  const goBack = useCallback(() => {
    if (indexRef.current <= 0) return;

    isInternalNavRef.current = true;
    const newIndex = indexRef.current - 1;
    indexRef.current = newIndex;

    setCanGoBack(newIndex > 0);
    setCanGoForward(true);

    router.push(historyRef.current[newIndex]);
  }, [router]);

  const goForward = useCallback(() => {
    if (indexRef.current >= historyRef.current.length - 1) return;

    isInternalNavRef.current = true;
    const newIndex = indexRef.current + 1;
    indexRef.current = newIndex;

    setCanGoBack(true);
    setCanGoForward(newIndex < historyRef.current.length - 1);

    router.push(historyRef.current[newIndex]);
  }, [router]);

  return { canGoBack, canGoForward, goBack, goForward };
};
```

```tsx
    message: string;

      setSubmitResult(null);

          setSubmitResult({ type: 'success', message: t('successMessage') });
          reset();
        } else {
          console.error(ERROR_LOG.API, parsedData.error);
          setSubmitResult({
            type: 'error',
            message: parsedData.error || t('errorFallback'),
          });

        setSubmitResult({ type: 'error', message: t('errorNetwork') });
```

```tsx
'use client';

import { useRef, useState, HTMLAttributes } from 'react';
import { useTranslations } from 'next-intl';
import { CopyIcon, CheckIcon } from '@/components/icons/mdx';

const COPY_FEEDBACK_DURATION_MS = 1500;
const ERROR_COPY_FAILED = 'Failed to copy text: ';

type CodeBlockProps = HTMLAttributes<HTMLPreElement>;

export function CodeBlock({ children, ...props }: CodeBlockProps) {
  const t = useTranslations('common');
  const preRef = useRef<HTMLPreElement>(null);
  const [copied, setCopied] = useState(false);

  const handleCopy = async () => {
    const codeElement = preRef.current?.querySelector('code');
    const codeText = codeElement?.textContent || '';

    if (!codeText) return;

    try {
      await navigator.clipboard.writeText(codeText);
      setCopied(true);
      setTimeout(() => setCopied(false), COPY_FEEDBACK_DURATION_MS);
    } catch (err) {
      console.error(ERROR_COPY_FAILED, err);
    }
  };

  return (
    <div className='relative my-6 font-mono text-sm group code-block-wrapper not-prose'>
      <div className='bg-[var(--color-window-bg)] border-t-2 border-l-2 border-white border-r-2 border-r-[var(--color-border-dark)] border-b-2 border-b-[var(--color-border-dark)] px-2 py-1 flex justify-between items-center'>
```

```ts
import { describe, it, expect } from 'vitest';
import en from '../../../messages/en.json';
import ko from '../../../messages/ko.json';

function extractKeys(obj: Record<string, unknown>, prefix = ''): string[] {
  return Object.entries(obj).flatMap(([key, value]) => {
    const fullKey = prefix ? `${prefix}.${key}` : key;
    if (typeof value === 'object' && value !== null && !Array.isArray(value)) {
      return extractKeys(value as Record<string, unknown>, fullKey);
    }
    return [fullKey];
  });
}

describe('i18n key parity', () => {
  it('[parity] en.json and ko.json must have identical key sets', () => {
    // Arrange
    const enKeys = extractKeys(en as Record<string, unknown>).sort();
    const koKeys = extractKeys(ko as Record<string, unknown>).sort();

    // Act + Assert
    const missingInKo = enKeys.filter((k) => !koKeys.includes(k));
    const missingInEn = koKeys.filter((k) => !enKeys.includes(k));

    expect(missingInKo, 'Keys in en.json missing from ko.json').toEqual([]);
    expect(missingInEn, 'Keys in ko.json missing from en.json').toEqual([]);
  });
});
```
